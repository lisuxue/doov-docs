const USER_LINK0 = DOOV.string(DOOV.field<string, Model>('user', 'links', 0)); //access to the first element of iterable
const USER_LINK1 = DOOV.string(DOOV.field<string, Model>('user', 'links', 1)); //access to the second element of iterable
const USER_LINKS = DOOV.iterable<string>(DOOV.field<string[], Model>('user', 'links'))); //direct access to iterable

/* Multiple validations rule */
const multipleValidationsRule: ValidationRule = DOOV.validations(
      DOOV.when(USER_B.and(USER_ID.greaterOrEquals(2))).validate(),
      DOOV.when(USER_B.and(USER_NAME.endsWith('st'))).validate())

/* Multiple mappings Rule */
const multipleMappingRule: MappingRule = 
      DOOV.map(USER_ID.mapTo(StringFunction, v => 'link of ' + v)).to(USER_LINK1),
      DOOV.map(USER_NAME).to(USER_LINK2))
